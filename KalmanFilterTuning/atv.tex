\begin{frame}{O Filtro de Kalman Precisa de $Q$ e $R$}
    \begin{block}{Sistema Linear em Tempo Discreto}
        \[

    \end{block}
    \vspace{0.3cm}
    \begin{columns}[T]

\begin{frame}{Por Que $Q$ e $R$ Incorretos SĂ£o ProblemĂ¡ticos?}

\begin{frame}{Passos 1--2: Filtro de Kalman e AutocovariĂ¢ncias}
    \textbf{Passo 1 â€” Filtro com ganho constante $W$:}
    \[
        \hat{\mathbf{x}}(k{+}1|k) = F\,\hat{\mathbf{x}}(k|k), \qquad
        \boldsymbol{\nu}(k{+}1) = \mathbf{z}(k{+}1) - H\,\hat{\mathbf{x}}(k{+}1|k)
    \]
    \[
        \hat{\mathbf{x}}(k{+}1|k{+}1) = \hat{\mathbf{x}}(k{+}1|k) + W\,\boldsymbol{\nu}(k{+}1), \qquad
        \boldsymbol{\mu}(k{+}1) = \mathbf{z}(k{+}1) - H\,\hat{\mathbf{x}}(k{+}1|k{+}1)
    \]

    \vspace{0.3cm}
    \textbf{Passo 2 â€” AutocovariĂ¢ncia amostral (eq. 50):}
    \[
        \hat{C}(i) = \frac{1}{N-M}\sum_{j=1}^{N-M}\boldsymbol{\nu}(j)\,\boldsymbol{\nu}(j{+}i)',
        \qquad i = 0,1,\ldots,M{-}1
    \]

    \begin{alertblock}{ParĂ¢metro $M$ (nĂºmero de atrasos)}
        $M$ controla o compromisso entre variĂ¢ncia e resoluĂ§Ă£o temporal.
        Deve satisfazer $M \geq n_x$; tipicamente $M \in [40, 100]$.
    \end{alertblock}
\end{frame}

% ---------------------------------------------------------------------------
\begin{frame}{Passo 3: AtualizaĂ§Ă£o do Ganho $W$ por Descida de Gradiente}
    \textbf{Objetivo de brancura (eq. 52):}
    \[

            \sum_{i=1}^{M-1}
            \bigl[\text{diag}(\hat{C}(0))\bigr]^{-1/2}
            \hat{C}(i)'\,
            \bigl[\text{diag}(\hat{C}(0))\bigr]^{-1}

    $J = 0$ significa inovaĂ§Ăµes perfeitamente brancas $\Rightarrow$ filtro Ă³timo.

    \vspace{0.2cm}
    \textbf{Tamanho de passo adaptativo (Bold Driver, eq. 137):}
    \[

        \begin{cases}

                                     \quad\text{(passo muito grande)} \\
            \min\!\bigl(1{,}1\;\alpha^{(r-1)},\;\bar{c}\bigr)
                                  & \text{caso contrĂ¡rio}

    \]

    \begin{block}{CritĂ©rios de parada}

        $J < 10^{-6}$ \;|\; 5 iteraĂ§Ăµes sem melhora \;|\; $n_L$ iteraĂ§Ăµes mĂ¡ximas
    \end{block}
\end{frame}

\begin{frame}{Passo 4: EstimaĂ§Ă£o de $R$ (FĂ³rmula R3)}
    \textbf{ResĂ­duo pĂ³s-ajuste:}
    $\boldsymbol{\mu}(k) = (I - HW)\,\boldsymbol{\nu}(k)$
    \quad$\Rightarrow$\quad
    covariĂ¢ncia amostral: $\hat{G} = \tfrac{1}{N}\sum_k \boldsymbol{\mu}(k)\boldsymbol{\mu}(k)'$

    \vspace{0.3cm}
    \textbf{FĂ³rmula R3 â€” equaĂ§Ă£o de Riccati contĂ­nua (eq. 78):}
    \[

    \begin{columns}[T]
        \begin{column}{0.52\textwidth}
            \textbf{SoluĂ§Ă£o numĂ©rica (ApĂªndice F):}
            \begin{enumerate}
                \item Cholesky: $S^{-1} = \mathscr{L}\mathscr{L}'$
                \item EigendecomposiĂ§Ă£o: $\mathscr{L}'\hat{G}\mathscr{L} = U\Lambda U'$

        \end{column}
        \begin{column}{0.44\textwidth}

    \end{columns}
\end{frame}

\begin{frame}{Passos 5--6: EstimaĂ§Ă£o de $Q$/$P$ e LaĂ§o Externo}
    \textbf{Passo 5 â€” IteraĂ§Ă£o dupla (Lyapunov):}

    \smallskip
    \textit{InicializaĂ§Ă£o:} $\Gamma Q^{(0)}\Gamma' = WSW'$
